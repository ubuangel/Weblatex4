% This is a simple sample document.  For more complicated documents take a look in the exercise tab. Note that everything that comes after a % symbol is treated as comentario and ignored when the code is compiled.

\documentclass{article} % \documentclass{} is the first command in any LaTeX code.  It is used to define what kind of document you are creating such as an article or a book, and begins the document preamble

\usepackage{amsmath} % \usepackage is a command that allows you to add functionality to your LaTeX code

\title{Simple Sample} % Sets article title
\author{My Angel} % Sets authors name
\date{\today} % Sets date for date compiled

% The preamble ends with the command \begin{document}
\begin{document} % All begin commands must be paired with an end command somewhere
\maketitle % crear titulo preanbulo etc (title, author, date)

\section{Holamundillo de nuevo!} % creates a section


\begin{equation} % Creates an equation environment and is compiled as math
  \gamma^2+\theta^2=\omega^2
\end{equation}


\end{document} % This is the end of the document
